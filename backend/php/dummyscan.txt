    <script src="https://unpkg.com/jsqr@1.4.0/dist/jsQR.js"></script>

<body>
    <div class="scanner-container">
        <h1>ENTRY SCANNER</h1>
        <p class="scan-instruction">Position QR code within frame to scan</p>
        
        <div id="reader">
            <video id="preview" style="width: 100%; height: 100%; object-fit: cover;"></video>
        </div>
        <div id="result">Scan results will appear here...</div>

        <div class="buttons">
            <label class="btn btn-upload">
                Upload QR Image
                <input type="file" id="qr-input-file" accept="image/*">
            </label>
            <button id="send-btn" class="btn btn-submit" disabled onclick="sendToDatabase()">
                Submit Details
            </button>
        </div>
    </div>

    <script>
        let name = "", rollNumber = "", phoneNumber = "", carNumber = "N/A", scanTime = "";
        let scanner = null;
    
        function escapeHTML(text) {
            return text.replace(/</g, "&lt;").replace(/>/g, "&gt;");
        }

        function onScanSuccess(content) {
            try {
                console.log("Scanned Text:", content);

                let timeMatch = content.match(/(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2})/);
                let scanTimeUTC = timeMatch ? new Date(timeMatch[1]) : new Date();
                let scanTimeIST = new Date(scanTimeUTC.getTime() + (5.5 * 60 * 60 * 1000))
                                    .toISOString().slice(0, 19).replace('T', ' ');

                let nameMatch = content.match(/(?:Guest Name|Name)[:\s]*([A-Za-z\s]+)(?=\n|$)/i);
                let rollMatch = content.match(/(?:Roll Number|Student ID)[:\s]*([\w\d.-]+)/i);
                let phoneMatch = content.match(/(?:Phone Number|Phone)[:\s]*(\d{10})/);
                let carMatch = content.match(/(?:Car Number|Vehicle)[:\s]*([\w\d-]+)/i);

                name = nameMatch ? nameMatch[1].trim() : "N/A";
                rollNumber = rollMatch ? rollMatch[1].trim() : "N/A";
                phoneNumber = phoneMatch ? phoneMatch[1].trim() : "N/A";
                carNumber = carMatch ? carMatch[1].trim() : "N/A";
                scanTime = scanTimeIST;

                document.getElementById('result').innerHTML = `
                    <strong>Name:</strong> ${escapeHTML(name)}<br>
                    <strong>Student ID:</strong> ${escapeHTML(rollNumber)}<br>
                    <strong>Phone:</strong> ${escapeHTML(phoneNumber)}<br>
                    <strong>Car Number:</strong> ${escapeHTML(carNumber)}<br>
                    <strong>Scanned At (IST):</strong> ${escapeHTML(scanTime)}
                `;

                document.getElementById('send-btn').disabled = false;
            } catch (error) {
                document.getElementById('result').textContent = "Error scanning QR code";
                document.getElementById('send-btn').disabled = true;
            }
        }

        function sendToDatabase() {
            let formData = new FormData();
            formData.append("name", name);
            formData.append("roll_number", rollNumber);
            formData.append("phone_number", phoneNumber);
            formData.append("car_number", carNumber);
            formData.append("scan_time", scanTime);

            fetch("save_qr_data.php", { method: "POST", body: formData })
            .then(response => response.json())
            .then(data => {
                alert(data.status === "success" ? "Details submitted successfully" : "Error: " + data.message);
            })
            .catch(error => {
                alert("Submission failed: " + error.message);
            });
        }

        function initScanner() {
            scanner = new Instascan.Scanner({ video: document.getElementById('preview'), mirror: false });
            scanner.addListener('scan', onScanSuccess);

            Instascan.Camera.getCameras().then(cameras => {
                if (cameras.length > 0) scanner.start(cameras[0]);
                else throw new Error('No cameras found');
            }).catch(e => {
                document.getElementById('result').textContent = "Camera error: " + e;
            });
        }

        document.getElementById("qr-input-file").addEventListener("change", function(event) {
            let file = event.target.files[0];
            let reader = new FileReader();
            reader.onload = function(e) {
                let img = new Image();
                img.onload = function() {
                    let canvas = document.createElement('canvas');
                    let ctx = canvas.getContext('2d');
                    canvas.width = img.width;
                    canvas.height = img.height;
                    ctx.drawImage(img, 0, 0);
                    let qr = jsQR(ctx.getImageData(0, 0, canvas.width, canvas.height).data, canvas.width, canvas.height);
                    if (qr) onScanSuccess(qr.data);
                };
                img.src = e.target.result;
            };
            reader.readAsDataURL(file);
        });

        window.addEventListener('DOMContentLoaded', initScanner);
    </script>
</body>
